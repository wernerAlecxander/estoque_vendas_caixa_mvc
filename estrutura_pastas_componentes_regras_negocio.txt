│   │
│   ├── controllers/                    # [C] CONTROLLER: Lógica de negócio (Next.js Server Actions)

│   │   └── financeiro.controller.ts    # Roda as fórmulas (Ticket Médio, Margem Líquida)
│   │
│   └── app/                            # [V] VIEW: Roteamento e Componentes Visuais (Tailwind CSS)
